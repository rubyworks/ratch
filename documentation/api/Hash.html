argumentize(args_field=nil) click to toggle source

Turn a hash into arguments. 

  h = { :list => [1,2], :base => "HI" }
  h.argumentize #=> [ [], { :list => [1,2], :base => "HI" } ]
  h.argumentize(:list) #=> [ [1,2], { :base => "HI" } ]
